# ============================================================================
# RUNTIME — обязательные зависимости
# ============================================================================

# Основной UI-фреймворк. Используется во всех модулях:
#   QtWidgets, QtCore, QtGui, QtMultimedia (для QSoundEffect в mambo-режиме).
# PyQt6-Qt6 и PyQt6-sip ставятся автоматически как транзитивные зависимости.
PyQt6>=6.6,<7.0

# ============================================================================
# RUNTIME — опциональные зависимости
# ============================================================================

# Один из звуковых бэкендов для скрытой mambo-пасхалки (advanced_frame.py).
# Импорт обёрнут в try/except — без pygame приложение работает, просто
# не воспроизводит звук в этом режиме. Если ставить без pygame, бэкенд
# деградирует к winsound (Windows) или QSoundEffect (Qt).
pygame>=2.5,<3.0

# ============================================================================
# BUILD — нужны только для сборки .exe, не для запуска из исходников
# ============================================================================

pyinstaller>=6.0,<7.0
